    <h1 style="color:var(--red); font-size: 40px;">UJIAN TERKUNCI</h1>

<script>

    const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbwOJU0dGs4kZdHlP0As_rqMD7PtUdV6bGX8nORD8gdp/dev";

    let uRow, pCode, vCount = 0;

 

    async function handleLogin() {

        const u = document.getElementById('user').value;

        const p = document.getElementById('pass').value;

        document.getElementById('btn-login').innerText = "Verifikasi...";

        

        try {

            const res = await fetch(`${SCRIPT_URL}?action=login&user=${u}&pass=${p}`);

            const d = await res.json();

            if(d.status === "success") {

                uRow = d.row; pCode = d.pCode; vCount = d.violation;

                if(vCount >= 2) return forceLock();

                document.getElementById('login-page').style.display = 'none';

                showMapels();

            } else { throw "error"; }

        } catch(e) {

            document.getElementById('err').style.display = 'block';

            document.getElementById('btn-login').innerText = "Login";

        }

    }

 

    async function showMapels() {

        document.getElementById('mapel-page').style.display = 'flex';

        const res = await fetch(`${SCRIPT_URL}?action=getMapel`);

        const maps = await res.json();

        const container = document.getElementById('list-mapel');

        container.innerHTML = "";

        maps.forEach(m => {

            const b = document.createElement('button');

            b.className = "btn"; b.innerText = m.n;

            b.onclick = () => startExam(m.n, m.u);

            container.appendChild(b);

        });

    }

 

    function startExam(n, u) {

        document.getElementById('mapel-page').style.display = 'none';

        document.getElementById('exam-page').style.display = 'grid';

        document.getElementById('active-title').innerText = n;

        document.getElementById('exam-frame').src = u;

        

        document.documentElement.requestFullscreen().catch(() => {});

        startTimer(90);

        setupSecurity();

        for(let i=1; i<=40; i++) {

            const d = document.createElement('div');

            d.style = "border:1px solid #ccc; text-align:center; padding:8px; border-radius:4px; font-size:12px; font-weight:bold;";

            d.innerText = i;

            document.getElementById('q-grid').appendChild(d);

        }

    }

 

    function startTimer(m) {

        let s = m * 60;

        setInterval(() => {

            let mins = Math.floor(s/60); let secs = s%60;

            document.getElementById('timer-display').innerText = `${mins}:${secs<10?'0':''}${secs}`;

            if(s-- <= 0) { alert("Waktu Habis!"); window.location.reload(); }

        }, 1000);

    }

 

    function setupSecurity() {

        // Deteksi Keluar (Blur)

        window.onblur = () => {

            vCount++;

            fetch(`${SCRIPT_URL}?action=updateV&row=${uRow}&count=${vCount}`);

            if(vCount >= 2) forceLock();

            else alert("Peringatan! Jangan keluar dari aplikasi.");

        };

        // Blokir Klik Kanan & Keyboard

        document.addEventListener('contextmenu', e => e.preventDefault());

        document.onkeydown = (e) => { if(e.ctrlKey && 'cvu'.includes(e.key)) return false; };

    }

 

    function forceLock() { document.getElementById('lock-screen').style.display = 'block'; }

 

    function unlock() {

        if(document.getElementById('p-input').value === pCode) {

            vCount = 0;

            fetch(`${SCRIPT_URL}?action=updateV&row=${uRow}&count=0`);

            document.getElementById('lock-screen').style.display = 'none';

            document.documentElement.requestFullscreen();

        } else { alert("Kode Proktor Salah!"); }

    }

 

    function requestExit() {

        if(prompt("Masukkan Kode Keluar:") === "1234") window.location.reload();

    }

</script>
